  • grep
    • grep -rni pat # r for recursive, n for line Number, i for case Insensitive
    • grep -A 3 pat # -A 3 to show extra 3 lines Above, also -B 3 for Below, and -C 3 for Context, i.e. 3 lines Below and Above
    • grep -v pat # v for reVerse, i.e. all lines except the matching lines
  • find
    • find . -type f -name '*.py' -exec grep -I --exclude='tmp*.py' "pattern" {} + # exec grep "pattern" on each file with its name matching '*.py',
      • -I: Ignore binary files
      • --exclude='tmp*.py': exclude temp files with names like tmp1.py, tmp2.py
      • {}: brace expansion. More examples of brace/bracket expansion: echo b{ar,olt,ed}s see book "learning the bash shell"
  • diff
    • diff -u file1.txt file2.txt # u: Unified -> more concise
    • diff -c file1.txt file2.txt # c: Context
    • diff --side-by-side file1.txt file2.txt # diff output will be shown side by side
    • diff --ignore-white-space file1.txt file2.txt # white space diff will be ignored
    • diff --strip-trailing-cr file1.txt file2.txt # ignore carriage return characters (\r), which are common in files created on Windows (with CRLF line endings).
